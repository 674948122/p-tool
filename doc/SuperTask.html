function getUserinfo(params) {
    return new Promise((resolve, reject) => {
       setTimeout(() => {
           if (params.id == 2) {
               reject();
               return;
           }
           resolve({
               name: "张三",
               age: 18,
              ...params,
           });
       }, 2000);
   });
}
